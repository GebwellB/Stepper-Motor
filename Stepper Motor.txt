from machine import Pin
import utime
utime.sleep(0.1) # Wait for USB to become ready

def move_motor():
  global lowering
  while lowering <= 197:
    for step in step_sequence:
      for i in range(len(pins)):
        pins[i].value(step[i])
        utime.sleep(0.01)
        lowering += 1
        print(lowering)
  print("Sleeping")
  utime.sleep(0.1)
  while lowering >= 0:
    print("Ok")
    utime.sleep(2)
          

# Define pins
pins = [
  Pin(2, Pin.OUT),
  Pin(3, Pin.OUT),
]

# Create a sequence
step_sequence = [
  [1,0],
  [0,1],
]

lowering = 0

# Use loops to set pins in sequence
while True:
  move_motor()
